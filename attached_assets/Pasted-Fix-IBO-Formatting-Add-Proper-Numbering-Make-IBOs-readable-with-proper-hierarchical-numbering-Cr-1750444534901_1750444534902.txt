Create IBO parser and formatter:

javascript// /shared/iboFormatter.js
export const formatIBOsWithNumbering = (rawAIContent) => {
  const lines = rawAIContent.split('\n').filter(line => line.trim());
  const formatted = [];
  let currentBO = 0;
  let currentPM = 0;
  let currentOB = 0;
  let currentLO = 0;
  
  lines.forEach(line => {
    const trimmed = line.trim();
    
    // Business Objective (### Business Objective or similar)
    if (trimmed.match(/^#{1,4}\s*Business\s*Objective/i) || 
        trimmed.match(/^#{1,4}\s*BO\d*/i)) {
      currentBO++;
      currentPM = 0;
      formatted.push({
        type: 'business_objective',
        number: `BO${currentBO}`,
        title: extractTitle(trimmed),
        level: 1
      });
    }
    
    // WIIFM
    else if (trimmed.match(/^#{1,6}\s*WIIFM/i) || 
             trimmed.includes('What\'s In It For Me')) {
      formatted.push({
        type: 'wiifm',
        content: extractContent(trimmed),
        level: 2
      });
    }
    
    // Performance Metrics
    else if (trimmed.match(/^#{1,6}\s*Performance\s*Metric/i) ||
             trimmed.match(/^\d+\.\s/) ||
             trimmed.includes('Increase in') ||
             trimmed.includes('Improvement in')) {
      currentPM++;
      currentOB = 0;
      formatted.push({
        type: 'performance_metric',
        number: `${currentBO}.${currentPM}`,
        title: extractTitle(trimmed),
        level: 2
      });
    }
    
    // Observable Behaviors
    else if (trimmed.match(/Observable\s*Behavior/i) ||
             trimmed.match(/^[-*]\s*\*\*/) ||
             (trimmed.startsWith('-') && !trimmed.includes('Learning'))) {
      currentOB++;
      currentLO = 0;
      formatted.push({
        type: 'observable_behavior',
        number: `${currentBO}.${currentPM}.${currentOB}`,
        title: extractTitle(trimmed),
        level: 3
      });
    }
    
    // Learning Objectives
    else if (trimmed.match(/Learning\s*Objective/i) ||
             trimmed.match(/^[-*]\s*\*\*.*Learning/) ||
             (trimmed.includes('**') && trimmed.includes('Learning'))) {
      currentLO++;
      formatted.push({
        type: 'learning_objective',
        number: `${currentBO}.${currentPM}.${currentOB}.${currentLO}`,
        title: extractTitle(trimmed),
        level: 4
      });
    }
  });
  
  return formatted;
};

const extractTitle = (line) => {
  return line
    .replace(/^#{1,6}\s*/, '')
    .replace(/^\d+\.\s*/, '')
    .replace(/^[-*]\s*/, '')
    .replace(/\*\*/g, '')
    .replace(/Observable\s*Behaviors?\*\*\s*/i, '')
    .replace(/Learning\s*Objectives?\*\*\s*/i, '')
    .trim();
};
