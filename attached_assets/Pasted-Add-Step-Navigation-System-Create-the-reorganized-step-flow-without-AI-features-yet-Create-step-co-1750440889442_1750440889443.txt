Add step navigation hook:

typescript// /client/src/hooks/useSessionSteps.js
export const useSessionSteps = (sessionId) => {
  const [session, setSession] = useState(null);
  
  const updateStep = async (newStep) => {
    try {
      const response = await api.updateSessionProgress(sessionId, newStep);
      setSession(response.data);
    } catch (error) {
      console.error('Failed to update step:', error);
    }
  };
  
  const canAdvanceToStep = (stepId) => {
    const stepIndex = SESSION_STEPS.findIndex(s => s.id === stepId);
    const currentIndex = SESSION_STEPS.findIndex(s => s.id === session?.currentStep);
    return stepIndex <= currentIndex + 1;
  };
  
  return { session, updateStep, canAdvanceToStep };
};
